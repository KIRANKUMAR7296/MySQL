# `Mathematical` Functions:

```mysql
# Absolute:
SELECT ABS(-7)

# Ceiling:
SELECT CEILING(12.3); # 13

# FLOOR:
SELECT FLOOR(12.6); # 12

# Round:
SELECT ROUND(12.5); # 13
SELECT ROUND(12.4); # 12

# Truncate:
SELECT TRUNCATE(4567.6758, 2); # 4567.67
SELECT TRUNCATE(4567.6758, 2); # 4500

# Power
SELECT POWER(8, 2); # Square of 8: 64
SELECT POW(8,2);

# Square Root:
SELECT SQRT(64); # 8
SELECT SQRT(4086, 1/4) # Fourth Root: 8

# Random Number:
SELECT RAND();

# Degree:
SELECT DEGREES(PI()); # 180

# Radians:
SELECT RADIANS(180); # 3.14
```
